cmake_minimum_required(VERSION 3.2)

# set the project name
project(hestia-roguelike VERSION 1.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

add_subdirectory(depends/hge)

file(GLOB all_SRCS
        "${PROJECT_SOURCE_DIR}/src/*/*/*/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/**/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*/*.c"
        "${PROJECT_SOURCE_DIR}/src/*/*.c"
        "${PROJECT_SOURCE_DIR}/src/*.c"
        )

set (INCLUDE_DIRS
        include
        depends/hge/include)

include_directories(${INCLUDE_DIRS})
add_executable(hestia-Roguelike.app ${all_SRCS} include/world/map_grid.h include/world/floor_tile.h include/world/dungeonGenerator/dungeon_generator.h include/maths/delaunay.h include/maths/components/polygon.h include/maths/minimum_spanning_tree.h include/maths/components/triangulation.h include/maths/components/triangle.h include/maths/components/edge.h include/maths/components/vertex.h include/maths/components/graph.h include/maths/components/connection.h include/maths/components/subtree.h include/world/dungeonGenerator/dungeon_grid.h include/world/dungeonGenerator/grid_tile.h include/world/dungeonGenerator/room.h include/world/dungeonGenerator/path.h)
target_link_libraries(hestia-Roguelike.app hestia)
