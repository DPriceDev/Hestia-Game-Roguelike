cmake_minimum_required(VERSION 3.2)

# set the project name
project(hestia-roguelike VERSION 1.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fsanitize=address -fsanitize=undefined")

add_subdirectory(depends/hge)

file(GLOB all_SRCS
        "${PROJECT_SOURCE_DIR}/src/*/*/*/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/**/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*/*.c"
        "${PROJECT_SOURCE_DIR}/src/*/*.c"
        "${PROJECT_SOURCE_DIR}/src/*.c"
        )

set (INCLUDE_DIRS
        include
        depends/hge/include)

include_directories(${INCLUDE_DIRS})
add_executable(hestia-Roguelike.app ${all_SRCS} include/world/dungeon/room_generator.h include/world/dungeon/path_generator.h src/world/dungeon/room_generator.cpp src/world/dungeon/path_generator.cpp)
target_link_libraries(hestia-Roguelike.app hestia)
