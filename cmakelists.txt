cmake_minimum_required(VERSION 3.2)

# set the project name
project(hestia-roguelike VERSION 1.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

add_subdirectory(depends/hge)

file(GLOB all_SRCS
        "${PROJECT_SOURCE_DIR}/src/*/*/*/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/**/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*.cpp"
        "${PROJECT_SOURCE_DIR}/src/*/*/*.c"
        "${PROJECT_SOURCE_DIR}/src/*/*.c"
        "${PROJECT_SOURCE_DIR}/src/*.c"
        )

set (INCLUDE_DIRS
        include
        depends/hge/include)

include_directories(${INCLUDE_DIRS})
add_executable(hestia-Roguelike.app ${all_SRCS} include/map_grid.h include/floor_tile.h include/dungeon_generator.h include/delaunay.h include/delaunay_types.h)
target_link_libraries(hestia-Roguelike.app hestia)
